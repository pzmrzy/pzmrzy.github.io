    Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

    Report:
    dfs, record current string and how many parentheses already used
    Code:
      
    class Solution(object):
        def generateParenthesis(self, n):
            """
            :type n: int
            :rtype: List[str]
            """
            res = []
            q = [("", 0, 0)]
            while len(q) > 0:
                pat, l, r = q.pop()
                if l + r == n * 2:
                    res.append(pat)
                if l < n:
                    q.append( (pat + '(', l + 1, r) )
                if l > r:
                    q.append( (pat + ')', l, r + 1) )
            return res
