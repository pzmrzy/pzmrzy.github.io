    Given a non-negative integer represented as a non-empty array of digits, plus one to the integer.

    Report:
    if the result of the last bit is 10, carry become true, and continue, stop when carry is false
    Code:
      
    class Solution(object):
        def plusOne(self, digits):
            """
            :type digits: List[int]
            :rtype: List[int]
            """
            carry = False
            for i in range(len(digits)-1, -1, -1):
                if digits[i] + 1 > 9:
                    digits[i] = 0
                    carry = True
                else:
                    digits[i] = digits[i] + 1
                    carry = False
                    break
            if carry:
                return [1] + digits
            else:
                return digits
