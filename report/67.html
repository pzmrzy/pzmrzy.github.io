    Given two binary strings, return their sum

    Report:
    Make a and b same length by fill 0, from the last bit to do add, and recode carry.
    Code:
      
    class Solution(object):
        def addBinary(self, a, b):
            """
            :type a: str
            :type b: str
            :rtype: str
            """
            if a == '0' and b == '0':
                return '0'
            la, lb = len(a), len(b)
            if la > lb:
                b = '0' * (la - lb) + b
            else:
                a = '0' * (lb - la) + a
            A = map(int, a)
            B = map(int, b)
            l = len(A)
            res = [0] * (l + 1)
            c = 0
            for i in range(l, 0, -1):
                t = A[i - 1] + B[i - 1] + c
                c = t / 2
                res[i] = t % 2
            res[0] = c
            return ''.join(map(str, res)).lstrip('0')
